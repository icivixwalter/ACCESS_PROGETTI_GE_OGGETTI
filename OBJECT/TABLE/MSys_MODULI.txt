						INSERT INTO 
						MSys_ELENCO_ObjProgetti ( Scel_b, 
									  Name1_s,
									  IdOrd_lng, 
									  NRO_OGGETTO_i,
									  TIPOGGETTO_s,
									  COD_PROGETTO_s, 
									  PROGETTO_s,

						MSys_MODULI.Scel_b, 
						MSys_MODULI.Name1_s,
